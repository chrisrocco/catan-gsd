       ```typescript
       import { io as createClient, type Socket } from 'socket.io-client';
       import type { AddressInfo } from 'node:net';
       import { build } from '../index.js';
       import type { ServerToClientEvents, ClientToServerEvents } from '../types.js';

       export type TestClient = Socket<ServerToClientEvents, ClientToServerEvents>;

       export async function createTestServer() {
         const app = await build();
         await app.listen({ port: 0 });  // OS picks port
         const port = (app.server.address() as AddressInfo).port;
         return { app, port };
       }

       export function connectClient(port: number): TestClient {
         return createClient(`http://localhost:${port}`, {
           transports: ['websocket'],
           autoConnect: false,
         });
       }

       // Helper: connect and join a room in one step
       export function joinRoom(client: TestClient, code: string, displayName: string): Promise<{ ok: boolean; playerId?: string; error?: string }> {
         return new Promise((resolve) => {
           client.emit('join-room', { code, displayName }, resolve);
         });
       }

       // Helper: wait for next event of a specific type
       export function waitForEvent<K extends keyof ServerToClientEvents>(
         client: TestClient, event: K, timeout = 2000
       ): Promise<Parameters<ServerToClientEvents[K]>[0]> {
         return new Promise((resolve, reject) => {
           const timer = setTimeout(() => reject(new Error(`Timeout waiting for ${event}`)), timeout);
           client.once(event as string, (data: any) => {
             clearTimeout(timer);
             resolve(data);
           });
         });
       }
       ```

    2. **Create `packages/server/src/socket/lobbyHandlers.ts`**:
       - Export `registerLobbyHandlers(io: Server, socket: TypedSocket): void`
       - Use typed Socket from Socket.IO generics (Server<ClientToServerEvents, ServerToClientEvents, object, SocketData>)

       **`join-room` handler:**
       - Receive `{ code, displayName }` + acknowledgment callback
       - Look up room in roomStore; if not found, call callback `{ ok: false, error: 'Room not found' }`, return
       - If room is full (players.length + botCount >= 4), callback `{ ok: false, error: 'Room is full' }`, return
       - If game already started, callback `{ ok: false, error: 'Game already started' }`, return
       - Generate playerId via `crypto.randomUUID()`
       - Assign color: first available from `['red', 'blue', 'white', 'orange']` not taken by existing players
       - Determine isHost: `session.players.length === 0` (first joiner is host)
       - Call `session.addPlayer(socket.id, playerId, displayName, color, isHost)`
       - Set socket.data: roomCode, playerId, displayName, isHost
       - `socket.join(code)` — join Socket.IO room
       - Call callback `{ ok: true, playerId }`
       - Broadcast `lobby:state` to all in room (build LobbyState from session)

       **`set-bot-count` handler:**
       - Check socket.data.isHost; if not host, emit `room:error` and return
       - Validate count is 0-3 and players.length + count <= 4
       - Set `session.botCount = count`
       - Broadcast `lobby:state` to all in room

       **`start-game` handler:**
       - Check socket.data.isHost; if not host, emit `room:error` and return
       - Check at least 1 human player + botCount >= 2 total (minimum for Catan)
       - Call `session.startGame()`
       - Broadcast `game:state` to each player individually with filtered state (per-player loop using `session.filterStateFor(playerId)`)
       - For bot players, no socket exists — they do not receive state (Phase 3 will add bot turn logic)

       **`disconnecting` handler (NOT `disconnect`):**
       - Read `socket.data.roomCode` (still available in `disconnecting`, NOT in `disconnect`)
       - Remove player from session
       - If was host, call `session.promoteNextHost()` and update the new host's socket.data
       - If room is now empty AND game not started, delete from roomStore
       - Broadcast updated `lobby:state` to remaining players
