```typescript
export interface Hex {
  key: string;       // "q,r,s" e.g. "0,0,0"
  q: number; r: number; s: number;
  resource: ResourceType | null;  // null = desert
  number: number | null;          // null = desert
  vertexKeys: string[];  // 6 vertex keys
  edgeKeys: string[];    // 6 edge keys
}

export interface Vertex {
  key: string;           // "h1|h2|h3" sorted hex keys
  building: Building | null;
  port: Port | null;
  adjacentVertexKeys: string[];
  adjacentEdgeKeys: string[];
}

export interface Edge {
  key: string;           // "h1~h2" or "h1~sea~idx"
  road: Road | null;
  vertexKeys: [string, string];  // exactly 2 vertex endpoints
}

export interface Board {
  hexes: Record<string, Hex>;
  vertices: Record<string, Vertex>;
  edges: Record<string, Edge>;
}

export interface Building { playerId: string; type: BuildingType; }
export interface Road { playerId: string; }
export interface Port { type: PortType; }
export type PortType = '3:1' | ResourceType;
export type BuildingType = 'settlement' | 'city';
```

  <action>
  1. Create `packages/client/src/components/board/hexMath.ts`:

     Constants:
     - `HEX_SIZE = 50` (circumradius — center to vertex)
     - `SQRT3 = Math.sqrt(3)`

     Functions:
     - `cubeToPixel(q, r): { x, y }` — flat-top formula:
       `x = HEX_SIZE * 3/2 * q`
       `y = HEX_SIZE * (SQRT3/2 * q + SQRT3 * r)`

     - `hexPoints(cx, cy, size): string` — returns SVG polygon points string.
       Use flat-top angles [0, 60, 120, 180, 240, 300] degrees.
       Each point: `cx + size * cos(angle_rad), cy + size * sin(angle_rad)`.

     - `vertexPosition(vertexKey: string): { x, y }` — parse the vertex key (split by `|`),
       for each hex key part: split by `,` to get q,r,s numbers, compute cubeToPixel(q,r),
       then return centroid (average x, average y) of all parsed hex positions.
       This handles both interior vertices (3 hex keys) and border vertices (3 hex keys where
       some are virtual sea hexes — still valid for coordinate math).

     - `edgeEndpoints(edge: Edge, vertexPositions: Map<string, {x,y}>): [{x,y}, {x,y}]` —
       look up the edge's two vertexKeys in the vertex positions map.
       Return the two endpoint positions.

     - `parseCubeCoords(hexKey: string): { q: number, r: number, s: number }` —
       split by `,`, parse ints.

     Export all functions and constants.

  2. Create `packages/client/src/components/board/HexTile.tsx`:
     Props: `hex: Hex`, `cx: number`, `cy: number`, `size: number`

     Renders:
     - `<polygon>` with points from hexPoints(), fill from RESOURCE_COLORS[hex.resource] or DESERT_COLOR
     - Stroke: dark brown `#5a4a3a`, strokeWidth 2
     - If hex has a number, render `<NumberToken>` inside

  3. Create `packages/client/src/components/board/NumberToken.tsx`:
     Props: `cx: number`, `cy: number`, `number: number`

     Constants — pip counts per number token:
     ```
     TOKEN_PIPS: Record<number, number> = {
       2: 1, 3: 2, 4: 3, 5: 4, 6: 5,
       8: 5, 9: 4, 10: 3, 11: 2, 12: 1,
     }
     ```

     Renders:
     - White/cream circle (`#f5f0e8`) centered on (cx, cy), radius ~14, thin stroke
     - Number text: bold, centered, fontSize 14
     - Red fill (`#e74c3c`) for numbers 6 and 8, dark (`#333`) otherwise
     - Pip dots below the number: small circles (r=1.5) evenly spaced horizontally
     - Pip dot color matches text color (red for 6/8)

  4. Create `packages/client/src/components/board/PortLabel.tsx`:

     - Small background rect for readability
     - Position slightly offset from vertex toward the board edge
